<span data-ttu-id="6305c-150">請使用 [ForwardedHeadersOptions](/dotnet/api/microsoft.aspnetcore.builder.forwardedheadersoptions) 來設定中介軟體以轉送 `Startup.ConfigureServices` 中的 `X-Forwarded-For` 和 `X-Forwarded-Proto` 標頭。</span><span class="sxs-lookup"><span data-stu-id="6305c-150">Configure the middleware with [ForwardedHeadersOptions](/dotnet/api/microsoft.aspnetcore.builder.forwardedheadersoptions) to forward the `X-Forwarded-For` and `X-Forwarded-Proto` headers in `Startup.ConfigureServices`.</span></span> <span data-ttu-id="6305c-151">在呼叫其他中介軟體之前，請先在 `Startup.Configure` 中叫用 [UseForwardedHeaders](/dotnet/api/microsoft.aspnetcore.builder.forwardedheadersextensions.useforwardedheaders) 方法：</span><span class="sxs-lookup"><span data-stu-id="6305c-151">Invoke the [UseForwardedHeaders](/dotnet/api/microsoft.aspnetcore.builder.forwardedheadersextensions.useforwardedheaders) method in `Startup.Configure` before calling other middleware:</span></span>

    services.AddMvc();
    
    services.Configure<ForwardedHeadersOptions>(options =>

        options.ForwardedHeaders =

    });

public void Configure(IApplicationBuilder app, IHostingEnvironment env)

    app.UseForwardedHeaders();

    if (env.IsDevelopment())

    {

    }
